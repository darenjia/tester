```python
# canoe_com_wrapper.py
class CANoeCOMWrapper:
    """CANoe COM接口包装器"""
    
    def __init__(self):
        self._app = None
        self._measurement = None
        self._system = None
        self._namespaces = None
        
    def connect(self) -> bool:
        """连接CANoe应用程序"""
        try:
            import win32com.client
            self._app = win32com.client.Dispatch("CANoe.Application")
            self._measurement = self._app.Measurement
            self._system = self._app.System
            self._namespaces = self._system.Namespaces
            return True
        except Exception as e:
            logger.error(f"连接CANoe失败: {e}")
            return False
    
    def open_configuration(self, config_path: str) -> bool:
        """打开配置文件"""
        try:
            self._app.Open(config_path)
            return self._app.Configuration.OpenConfigurationResult == 0
        except Exception as e:
            logger.error(f"打开配置失败: {e}")
            return False
    
    def start_measurement(self) -> bool:
        """启动测量"""
        try:
            if not self._measurement.Running:
                self._measurement.Start()
            return True
        except Exception as e:
            logger.error(f"启动测量失败: {e}")
            return False
    
    def stop_measurement(self) -> bool:
        """停止测量"""
        try:
            if self._measurement.Running:
                self._measurement.Stop()
            return True
        except Exception as e:
            logger.error(f"停止测量失败: {e}")
            return False
    
    def get_signal_value(self, signal_name: str, bus: str = "CAN") -> float:
        """获取信号值"""
        try:
            signal = self._app.GetBus(bus).GetSignal(0, signal_name)
            return signal.Value
        except Exception as e:
            logger.error(f"获取信号失败: {e}")
            return None
    
    def set_signal_value(self, signal_name: str, value: float, bus: str = "CAN") -> bool:
        """设置信号值"""
        try:
            signal = self._app.GetBus(bus).GetSignal(0, signal_name)
            signal.Value = value
            return True
        except Exception as e:
            logger.error(f"设置信号失败: {e}")
            return False
    
    def get_system_variable(self, namespace: str, variable: str):
        """获取系统变量值"""
        try:
            ns = self._namespaces[namespace]
            var = ns.Variables[variable]
            return var.Value
        except Exception as e:
            logger.error(f"获取系统变量失败: {e}")
            return None
    
    def set_system_variable(self, namespace: str, variable: str, value):
        """设置系统变量值"""
        try:
            ns = self._namespaces[namespace]
            var = ns.Variables[variable]
            var.Value = value
            return True
        except Exception as e:
            logger.error(f"设置系统变量失败: {e}")
            return False
    
    def close(self):
        """关闭CANoe"""
        try:
            if self._measurement and self._measurement.Running:
                self._measurement.Stop()
            if self._app:
                self._app.Quit()
        except:
            pass
```

```python
# canoe_test_engine.py
class CANoeTestEngine:
    """CANoe测试执行引擎"""
    
    def __init__(self, canoe_wrapper: CANoeCOMWrapper):
        self._canoe = canoe_wrapper
        self._test_cache = []
        self._result_cache = {}
        self._is_running = False
        self._current_case = None
        
    def execute_self_check(self, config_path: str) -> Dict[str, Any]:
        """执行设备自检"""
        result = {
            "status": "running",
            "device_info": {},
            "errors": []
        }
        
        try:
            # 1. 打开自检配置
            if not self._canoe.open_configuration(config_path):
                result["status"] = "failed"
                result["errors"].append("打开自检配置失败")
                return result
            
            # 2. 启动测量
            self._canoe.start_measurement()
            
            # 3. 触发自检程序
            self._canoe.set_system_variable("mutualVar", "startDeviceSelfCheck", 1)
            
            # 4. 等待自检完成
            timeout = 300  # 5分钟超时
            start_time = time.time()
            while time.time() - start_time < timeout:
                is_end = self._canoe.get_system_variable("mutualVar", "isEndDeviceSelfCheck")
                if is_end == 1:
                    break
                time.sleep(0.5)
            
            # 5. 获取设备信息
            result["device_info"] = self._read_device_info()
            result["status"] = "completed"
            
        except Exception as e:
            result["status"] = "failed"
            result["errors"].append(str(e))
        
        return result
    
    def execute_test_cases(self, cases: List[Dict], dtc_info: Dict) -> List[Dict]:
        """执行测试用例"""
        results = []
        
        for case in cases:
            case_result = self._execute_single_case(case, dtc_info)
            results.append(case_result)
            
        return results
    
    def _execute_single_case(self, case: Dict, dtc_info: Dict) -> Dict:
        """执行单个用例"""
        case_name = case.get("name")
        case_type = case.get("type", "explicit")  # explicit/implicit
        
        result = {
            "name": case_name,
            "type": case_type,
            "status": "running",
            "start_time": datetime.now().isoformat(),
            "end_time": None,
            "result_code": None,
            "log": []
        }
        
        try:
            # 1. 设置DTC信息
            if case_name in dtc_info:
                self._canoe.set_system_variable(
                    "mutualVar", 
                    "dtcTestInformation", 
                    dtc_info[case_name]
                )
            
            # 2. 设置用例名称
            self._canoe.set_system_variable("mutualVar", "testScriptName", case_name)
            time.sleep(0.2)
            
            # 3. 触发测试
            self._canoe.set_system_variable("mutualVar", "startTest", 1)
            time.sleep(0.2)
            
            # 4. 等待测试完成
            timeout = case.get("timeout", 600)
            start_time = time.time()
            
            while time.time() - start_time < timeout:
                # 检查缓冲标志
                buffer_flag = self._canoe.get_system_variable("mutualVar", "bufferFlag")
                if buffer_flag == 1:
                    buffer_value = self._canoe.get_system_variable("mutualVar", "bufferValue")
                    result["log"].append(buffer_value)
                    self._canoe.set_system_variable("mutualVar", "bufferFlag", 0)
                
                # 检查测试结束
                end_test = self._canoe.get_system_variable("mutualVar", "endTest")
                if end_test == 1:
                    self._canoe.set_system_variable("mutualVar", "endTest", 0)
                    result_code = self._canoe.get_system_variable("mutualVar", "testCaseResultState")
                    result["result_code"] = result_code
                    result["status"] = "passed" if result_code == 0 else "failed"
                    break
                
                time.sleep(2)
            
            if result["status"] == "running":
                result["status"] = "timeout"
                
        except Exception as e:
            result["status"] = "error"
            result["error"] = str(e)
        
        result["end_time"] = datetime.now().isoformat()
        return result
    
    def _read_device_info(self) -> Dict[str, str]:
        """读取设备信息"""
        info_keys = [
            "carManufacturerECUHardwareNumber",
            "carManufacturerECUSoftware",
            "ECUBatchNumber",
            "ECUManufacturingDate",
            "softwareVersionNumber",
            "VINCode"
        ]
        
        device_info = {}
        for key in info_keys:
            try:
                value = self._canoe.get_system_variable("mutualVar", key)
                device_info[key] = value
            except:
                device_info[key] = ""
        
        return device_info
```

```python
# canoe_adapter.py
class CANoeAdapter(BaseTestAdapter):
    """
    CANoe测试工具适配器（完善版）
    
    通过Python直接调用CANoe COM接口，支持：
    - CANoe应用程序控制
    - 配置文件加载
    - 测量启停
    - 信号读写
    - 系统变量操作
    - 测试用例执行
    """
    
    def __init__(self, config: dict = None):
        super().__init__(config)
        
        self._canoe_wrapper = CANoeCOMWrapper()
        self._test_engine = CANoeTestEngine(self._canoe_wrapper)
        
        # 配置
        self.start_timeout = self.config.get("start_timeout", 30)
        self.stop_timeout = self.config.get("stop_timeout", 10)
        self.measurement_timeout = self.config.get("measurement_timeout", 3600)
        
    @property
    def tool_type(self) -> TestToolType:
        return TestToolType.CANOE
    
    def connect(self) -> bool:
        """连接CANoe"""
        try:
            self.status = AdapterStatus.CONNECTING
            self.logger.info("正在连接CANoe...")
            
            if self._canoe_wrapper.connect():
                self.status = AdapterStatus.CONNECTED
                self._clear_error()
                self.logger.info("CANoe连接成功")
                return True
            else:
                self._set_error("CANoe连接失败")
                return False
                
        except Exception as e:
            self._set_error(f"CANoe连接异常: {str(e)}")
            return False
    
    def disconnect(self) -> bool:
        """断开CANoe"""
        try:
            self._canoe_wrapper.close()
            self.status = AdapterStatus.DISCONNECTED
            self.logger.info("CANoe已断开")
            return True
        except Exception as e:
            self._set_error(f"断开CANoe失败: {str(e)}")
            return False
    
    def load_configuration(self, config_path: str) -> bool:
        """加载配置文件"""
        if not self.is_connected:
            self._set_error("CANoe未连接")
            return False
        
        try:
            self.logger.info(f"加载配置: {config_path}")
            return self._canoe_wrapper.open_configuration(config_path)
        except Exception as e:
            self._set_error(f"加载配置失败: {str(e)}")
            return False
    
    def start_test(self) -> bool:
        """启动测量"""
        if not self.is_connected:
            self._set_error("CANoe未连接")
            return False
        
        try:
            if self._canoe_wrapper.start_measurement():
                self.status = AdapterStatus.RUNNING
                return True
            return False
        except Exception as e:
            self._set_error(f"启动测量失败: {str(e)}")
            return False
    
    def stop_test(self) -> bool:
        """停止测量"""
        if not self.is_connected:
            return True
        
        try:
            self._canoe_wrapper.stop_measurement()
            self.status = AdapterStatus.CONNECTED
            return True
        except Exception as e:
            self._set_error(f"停止测量失败: {str(e)}")
            return False
    
    def execute_test_item(self, item: Dict[str, Any]) -> Dict[str, Any]:
        """
        执行测试项
        
        支持的测试项类型：
        - self_check: 设备自检
        - test_cases: 执行测试用例
        - signal_check: 信号检查
        - signal_set: 信号设置
        """
        item_type = item.get("type")
        item_name = item.get("name", "unnamed")
        
        self.logger.info(f"执行测试项: {item_name} (类型: {item_type})")
        
        try:
            if item_type == "self_check":
                return self._test_engine.execute_self_check(
                    item.get("config_path")
                )
            elif item_type == "test_cases":
                return self._test_engine.execute_test_cases(
                    item.get("cases", []),
                    item.get("dtc_info", {})
                )
            elif item_type == "signal_check":
                return self._execute_signal_check(item)
            elif item_type == "signal_set":
                return self._execute_signal_set(item)
            else:
                return {
                    "name": item_name,
                    "type": item_type,
                    "status": "error",
                    "error": f"不支持的测试项类型: {item_type}"
                }
        except Exception as e:
            return {
                "name": item_name,
                "type": item_type,
                "status": "error",
                "error": str(e)
            }
    
    def _execute_signal_check(self, item: Dict) -> Dict:
        """执行信号检查"""
        signal_name = item.get("signal_name")
        expected_value = item.get("expected_value")
        tolerance = item.get("tolerance", 0.01)
        bus = item.get("bus", "CAN")
        
        actual_value = self._canoe_wrapper.get_signal_value(signal_name, bus)
        passed = False
        if actual_value is not None and expected_value is not None:
            passed = abs(actual_value - expected_value) < tolerance
        
        return {
            "name": item.get("name"),
            "type": "signal_check",
            "signal_name": signal_name,
            "expected_value": expected_value,
            "actual_value": actual_value,
            "passed": passed,
            "status": "passed" if passed else "failed"
        }
    
    def _execute_signal_set(self, item: Dict) -> Dict:
        """执行信号设置"""
        signal_name = item.get("signal_name")
        value = item.get("value")
        bus = item.get("bus", "CAN")
        
        success = self._canoe_wrapper.set_signal_value(signal_name, value, bus)
        
        return {
            "name": item.get("name"),
            "type": "signal_set",
            "signal_name": signal_name,
            "value": value,
            "success": success,
            "status": "passed" if success else "failed"
        }
```

```csharp
public class TestExecutionService
{
    private readonly ProcCANoeTest _canoeTest;
    private readonly Dictionary<string, TestTask> _tasks;
    
    public TestExecutionService()
    {
        _canoeTest = new ProcCANoeTest();
        _tasks = new Dictionary<string, TestTask>();
    }
    
    public TestResult StartTest(StartTestRequest request)
    {
        // 1. 设置路径
        _canoeTest.GetName(
            request.SelfCheckPath,
            request.ConfigPath,
            request.StateMachinePath
        );
        
        // 2. 设备自检
        if (!_canoeTest.StartDeviceSelfCheck())
        {
            throw new Exception("设备自检失败");
        }
        
        // 3. 等待自检完成
        WaitForSelfCheck();
        
        // 4. 创建任务
        var taskId = Guid.NewGuid().ToString();
        var task = new TestTask
        {
            TaskId = taskId,
            Status = TestStatus.Running,
            StartTime = DateTime.Now
        };
        _tasks[taskId] = task;
        
        // 5. 异步执行测试
        Task.Run(() => ExecuteTestsAsync(request, task));
        
        return new TestResult { TaskId = taskId };
    }
    
    private async Task ExecuteTestsAsync(StartTestRequest request, TestTask task)
    {
        try
        {
            // 执行显性用例
            if (request.ExplicitTestCases?.Count > 0)
            {
                GloalVar.IsDPause = true;
                _canoeTest.StartDExampleCheck(
                    request.ExplicitTestCases,
                    request.DtcInfo,
                    false
                );
            }
            
            // 执行隐性用例
            if (request.ImplicitTestCases?.Count > 0)
            {
                GloalVar.IsHPause = true;
                _canoeTest.StartHExampleCheck(
                    request.ImplicitTestCases,
                    request.DtcInfo
                );
            }
            
            task.Status = TestStatus.Completed;
        }
        catch (Exception ex)
        {
            task.Status = TestStatus.Failed;
            task.ErrorMessage = ex.Message;
        }
    }
}
```

```python
class CanoeCSharpAdapter(BaseTestAdapter):
    """
    CANoe C#中间件适配器
    
    通过HTTP API调用C#中间层服务控制CANoe
    """
    
    def __init__(self, config: dict = None):
        super().__init__(config)
        self.base_url = config.get('csharp_middleware_url', 'http://localhost:5000')
        self.timeout = config.get('timeout', 3600)
        self._current_task_id = None
        
    @property
    def tool_type(self) -> TestToolType:
        return TestToolType.CANOE_CSHARP
    
    def connect(self) -> bool:
        """检查C#服务是否可用"""
        try:
            response = requests.get(f"{self.base_url}/api/health", timeout=5)
            return response.status_code == 200
        except:
            return False
    
    def execute_test_item(self, item: Dict[str, Any]) -> Dict[str, Any]:
        """执行测试项"""
        test_request = {
            'configPath': item.get('config_path'),
            'selfCheckPath': item.get('self_check_path'),
            'explicitTestCases': item.get('explicit_cases', []),
            'implicitTestCases': item.get('implicit_cases', []),
            'dtcInfo': item.get('dtc_info', {})
        }
        
        # 启动测试
        response = requests.post(
            f"{self.base_url}/api/test/start",
            json=test_request,
            timeout=30
        )
        result = response.json()
        
        if not result.get('success'):
            return {'status': 'error', 'error': result.get('error')}
        
        self._current_task_id = result['data']['taskId']
        
        # 轮询等待完成
        return self._wait_for_completion()
    
    def _wait_for_completion(self) -> Dict[str, Any]:
        """等待测试完成"""
        max_wait = self.timeout
        waited = 0
        
        while waited < max_wait:
            status = self.get_status()
            if not status.get('isRunning'):
                break
            time.sleep(2)
            waited += 2
        
        # 获取结果
        response = requests.get(
            f"{self.base_url}/api/test/result",
            params={"taskId": self._current_task_id}
        )
        return response.json().get('data', {})
```

```python
# adapters/__init__.py
class AdapterFactory:
    """适配器工厂（增强版）"""
    
    _registry = {
        TestToolType.CANOE: CANoeAdapter,
        TestToolType.CANOE_CSHARP: CanoeCSharpAdapter,
        TestToolType.TSMASTER: TSMasterAdapter,
        TestToolType.TTWORKBENCH: TTworkbenchAdapter,
    }
    
    _instances = {}
    
    @classmethod
    def create_adapter(cls, tool_type: TestToolType, config: dict = None, 
                       singleton: bool = True) -> BaseTestAdapter:
        """
        创建适配器实例
        
        Args:
            tool_type: 测试工具类型
            config: 配置字典
            singleton: 是否使用单例模式
            
        Returns:
            适配器实例
        """
        if tool_type not in cls._registry:
            raise ValueError(f"不支持的测试工具类型: {tool_type}")
        
        # 单例模式
        if singleton:
            key = f"{tool_type.value}_{hash(str(config))}"
            if key not in cls._instances:
                adapter_class = cls._registry[tool_type]
                cls._instances[key] = adapter_class(config)
            return cls._instances[key]
        
        # 新实例
        adapter_class = cls._registry[tool_type]
        return adapter_class(config)
    
    @classmethod
    def register_adapter(cls, tool_type: TestToolType, 
                         adapter_class: Type[BaseTestAdapter]):
        """注册新的适配器类型"""
        cls._registry[tool_type] = adapter_class
    
    @classmethod
    def get_available_adapters(cls) -> List[TestToolType]:
        """获取所有可用的适配器类型"""
        available = []
        for tool_type in cls._registry:
            try:
                adapter = cls.create_adapter(tool_type, singleton=False)
                if adapter.connect():
                    available.append(tool_type)
                    adapter.disconnect()
            except:
                pass
        return available
```

```python
# core/adapter_manager.py
class AdapterManager:
    """适配器管理器"""
    
    def __init__(self):
        self._adapters: Dict[str, BaseTestAdapter] = {}
        self._config_manager = AdapterConfigManager()
    
    def get_adapter(self, device_id: str, tool_type: TestToolType) -> BaseTestAdapter:
        """获取或创建适配器"""
        key = f"{device_id}_{tool_type.value}"
        
        if key not in self._adapters:
            config = self._config_manager.get_config(tool_type.value)
            adapter = AdapterFactory.create_adapter(tool_type, config)
            self._adapters[key] = adapter
        
        return self._adapters[key]
    
    def connect_device(self, device_id: str, tool_type: TestToolType) -> bool:
        """连接设备"""
        adapter = self.get_adapter(device_id, tool_type)
        return adapter.connect()
    
    def disconnect_device(self, device_id: str, tool_type: TestToolType) -> bool:
        """断开设备"""
        key = f"{device_id}_{tool_type.value}"
        if key in self._adapters:
            adapter = self._adapters[key]
            result = adapter.disconnect()
            del self._adapters[key]
            return result
        return True
    
    def disconnect_all(self):
        """断开所有设备"""
        for key, adapter in list(self._adapters.items()):
            try:
                adapter.disconnect()
            except:
                pass
        self._adapters.clear()
```
