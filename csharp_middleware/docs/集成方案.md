# C#中间件集成方案

## 1. 代码架构分析

### 1.1 现有代码结构

```
UltraANetT/
├── CANoeEngine/          # CANoe核心引擎
│   ├── ProcCANoe.cs      # CANoe基础操作封装
│   ├── ProcCANoeTest.cs  # 测试执行控制
│   ├── ProcSelfCheck.cs  # 自检功能
│   └── ProcExample.cs    # 用例执行
├── ProcessEngine/        # 业务处理引擎
│   ├── ProcFile.cs       # 文件处理
│   └── LogicalControl.cs # 逻辑控制
├── Model/                # 数据模型
└── DBEngine/             # 数据库引擎
```

### 1.2 核心功能模块

| 模块 | 功能 | 关键类 |
|------|------|--------|
| **CANoe操作** | 连接/断开CANoe、加载配置、启停测量 | `ProcCANoe` |
| **测试执行** | 执行测试用例、获取结果、状态监控 | `ProcCANoeTest` |
| **系统变量交互** | 与CANoe CAPL变量交互 | `ProcSelfCheck`, `ProcExample` |

| **自检功能** | 设备自检、样件自检 | `ProcSelfCheck` |

```csharp
// TestExecutionService.cs - 测试执行服务
public class TestExecutionService
{
    private readonly ProcCANoeTest _canoeTest;
    private readonly Dictionary<string, TestTask> _runningTasks;
    
    public TestExecutionService()
    {
        _canoeTest = new ProcCANoeTest();
        _runningTasks = new Dictionary<string, TestTask>();
    }
    
    // 启动测试
    public async Task<TestResult> StartTestAsync(StartTestRequest request)
    {
        // 1. 设置路径
        _canoeTest.GetName(
            request.SelfCheckPath, 
            request.ConfigPath, 
            request.StateMachinePath
        );
        
        // 2. 启动设备自检
        bool selfCheckResult = _canoeTest.StartDeviceSelfCheck();
        if (!selfCheckResult)
        {
            throw new Exception("设备自检失败");
        }
        
        // 3. 等待自检完成
        await WaitForSelfCheckCompleteAsync();
        
        // 4. 执行测试用例
        var taskId = Guid.NewGuid().ToString();
        var testTask = new TestTask
        {
            TaskId = taskId,
            Status = TestStatus.Running,
            StartTime = DateTime.Now
        };
        _runningTasks[taskId] = testTask;
        
        // 5. 异步执行测试
        _ = Task.Run(() => ExecuteTestCasesAsync(request, testTask));
        
        return new TestResult { TaskId = taskId };
    }
    
    // 执行测试用例
    private async Task ExecuteTestCasesAsync(StartTestRequest request, TestTask task)
    {
        try
        {
            // 显性用例测试
            if (request.ExplicitTestCases?.Count > 0)
            {
                GloalVar.IsDPause = true;
                int result = _canoeTest.StartDExampleCheck(
                    request.ExplicitTestCases,
                    request.DtcInfo,
                    false
                );
                
                if (result != 0)
                {
                    task.Status = TestStatus.Failed;
                    task.ErrorMessage = "显性用例测试失败";
                    return;
                }
            }
            
            // 隐性用例测试
            if (request.ImplicitTestCases?.Count > 0)
            {
                GloalVar.IsHPause = true;
                int result = _canoeTest.StartHExampleCheck(
                    request.ImplicitTestCases,
                    request.DtcInfo
                );
                
                if (result != 0)
                {
                    task.Status = TestStatus.Failed;
                    task.ErrorMessage = "隐性用例测试失败";
                    return;
                }
            }
            
            task.Status = TestStatus.Completed;
            task.EndTime = DateTime.Now;
        }
        catch (Exception ex)
        {
            task.Status = TestStatus.Failed;
            task.ErrorMessage = ex.Message;
        }
    }
    
    // 获取实时状态
    public TestStatus GetStatus(string taskId)
    {
        if (!_runningTasks.ContainsKey(taskId))
        {
            return null;
        }
        
        var task = _runningTasks[taskId];
        return new TestStatus
        {
            TaskId = taskId,
            IsRunning = task.Status == TestStatus.Running,
            CurrentCase = GetCurrentCaseName(),
            CompletedCases = _canoeTest.ExampleTestedCount,
            TotalCases = _canoeTest.ExampleAllCount,
            DeviceInfo = _canoeTest._dictDeviceInfo
        };
    }
    
    // 停止测试
    public void StopTest(string taskId)
    {
        _canoeTest.PauseCANoe();
        _canoeTest.CloseCANoe();
        
        if (_runningTasks.ContainsKey(taskId))
        {
            _runningTasks[taskId].Status = TestStatus.Stopped;
        }
    }
    
    // 获取测试结果
    public TestResult GetResult(string taskId)
    {
        var results = new List<CaseResult>();
        
        // 显性用例结果
        foreach (var item in _canoeTest._dictDNameList)
        {
            results.Add(new CaseResult
            {
                CaseNo = item.Key,
                Result = ConvertResultCode(item.Value),
                CaseType = "显性"
            });
        }
        
        // 隐性用例结果
        foreach (var item in _canoeTest._dictHNameList)
        {
            results.Add(new CaseResult
            {
                CaseNo = item.Key,
                Result = ConvertResultCode(item.Value),
                CaseType = "隐性"
            });
        }
        
        return new TestResult
        {
            TaskId = taskId,
            Results = results,
            ReportPath = GetReportPath()
        };
    }
    
    private string ConvertResultCode(int code)
    {
        return code switch
        {
            0 => "PASS",
            1 => "FAIL",
            _ => "UNKNOWN"
        };
    }
}
```

### 3.2 Python适配器实现

```python
# canoe_csharp_adapter.py
import requests
import json
from typing import Dict, List, Optional
from core.adapters.base_adapter import BaseTestAdapter

class CanoeCSharpAdapter(BaseTestAdapter):
    """CANoe C#中间件适配器"""
    
    def __init__(self, config: dict = None):
        super().__init__(config)
        self.base_url = config.get('csharp_middleware_url', 'http://localhost:5000')
        self.timeout = config.get('timeout', 3600)
        self._current_task_id = None
        
    def connect(self) -> bool:
        """检查C#中间件服务是否可用"""
        try:
            response = requests.get(f"{self.base_url}/api/health", timeout=5)
            return response.status_code == 200
        except:
            return False
    
    def disconnect(self) -> bool:
        """断开连接"""
        if self._current_task_id:
            self.stop_test()
        return True
    
    def load_configuration(self, config_path: str) -> bool:
        """加载配置文件"""
        try:
            response = requests.post(
                f"{self.base_url}/api/config/load",
                json={"configPath": config_path},
                timeout=10
            )
            return response.json().get('success', False)
        except Exception as e:
            self.logger.error(f"加载配置失败: {e}")
            return False
    
    def start_test(self, test_request: dict) -> dict:
        """启动测试"""
        try:
            response = requests.post(
                f"{self.base_url}/api/test/start",
                json=test_request,
                timeout=30
            )
            result = response.json()
            if result.get('success'):
                self._current_task_id = result['data']['taskId']
            return result
        except Exception as e:
            self.logger.error(f"启动测试失败: {e}")
            return {'success': False, 'error': str(e)}
    
    def stop_test(self) -> bool:
        """停止测试"""
        if not self._current_task_id:
            return True
            
        try:
            response = requests.post(
                f"{self.base_url}/api/test/stop",
                json={"taskId": self._current_task_id},
                timeout=10
            )
            return response.json().get('success', False)
        except Exception as e:
            self.logger.error(f"停止测试失败: {e}")
            return False
    
    def get_status(self) -> dict:
        """获取测试状态"""
        if not self._current_task_id:
            return {'isRunning': False}
            
        try:
            response = requests.get(
                f"{self.base_url}/api/test/status",
                params={"taskId": self._current_task_id},
                timeout=5
            )
            return response.json().get('data', {})
        except Exception as e:
            self.logger.error(f"获取状态失败: {e}")
            return {'isRunning': False, 'error': str(e)}
    
    def get_result(self) -> dict:
        """获取测试结果"""
        if not self._current_task_id:
            return {'results': []}
            
        try:
            response = requests.get(
                f"{self.base_url}/api/test/result",
                params={"taskId": self._current_task_id},
                timeout=10
            )
            return response.json().get('data', {})
        except Exception as e:
            self.logger.error(f"获取结果失败: {e}")
            return {'results': [], 'error': str(e)}
    
    def execute_test_item(self, item: dict) -> dict:
        """执行单个测试项"""
        # 构建测试请求
        test_request = {
            'configPath': item.get('configPath'),
            'selfCheckPath': item.get('selfCheckPath'),
            'explicitTestCases': item.get('explicitCases', []),
            'implicitTestCases': item.get('implicitCases', []),
            'dtcInfo': item.get('dtcInfo', {}),
            'timeout': item.get('timeout', self.timeout)
        }
        
        # 启动测试
        start_result = self.start_test(test_request)
        if not start_result.get('success'):
            return {
                'caseNo': item.get('caseNo'),
                'result': 'FAIL',
                'remark': f"启动失败: {start_result.get('error')}"
            }
        
        # 轮询等待测试完成
        import time
        max_wait = test_request['timeout']
        waited = 0
        
        while waited < max_wait:
            status = self.get_status()
            if not status.get('isRunning'):
                break
            time.sleep(2)
            waited += 2
        
        # 获取结果
        result = self.get_result()
        
        return {
            'caseNo': item.get('caseNo'),
            'result': 'PASS' if result.get('success') else 'FAIL',
            'remark': result.get('message', ''),
            'reportPath': result.get('reportPath')
        }
```

### 3.3 适配器工厂更新

```python
# core/adapters/__init__.py

from enum import Enum
from typing import Optional, Dict, Any

class TestToolType(Enum):
    CANOE = "canoe"
    CANOE_CSHARP = "canoe_csharp"  # C#中间件方式
    TSMASTER = "tsmaster"
    TTWORKBENCH = "ttworkbench"

ADAPTER_REGISTRY = {
    TestToolType.CANOE: CANoeAdapter,
    TestToolType.CANOE_CSHARP: CanoeCSharpAdapter,  # 新增
    TestToolType.TSMASTER: TSMasterAdapter,
    TestToolType.TTWORKBENCH: TTworkbenchAdapter,
}

def create_adapter(tool_type: TestToolType, config: dict = None) -> BaseTestAdapter:
    """创建适配器实例"""
    adapter_class = ADAPTER_REGISTRY.get(tool_type)
    if not adapter_class:
        raise ValueError(f"不支持的测试工具类型: {tool_type}")
    return adapter_class(config)
```

## 6. 风险评估与应对

| 风险 | 影响 | 应对措施 |
|------|------|----------|
| C#代码耦合度高 | 改造难度大 | 采用包装器模式，最小化改动 |
| CANoe COM接口稳定性 | 可能导致崩溃 | 增加异常处理和重试机制 |
| 网络通信延迟 | 影响实时性 | 本地部署优先，优化超时设置 |
| 多线程并发问题 | 数据混乱 | 增加任务隔离和锁机制 |
| 原有功能兼容性 | 功能丢失 | 完整回归测试 |
